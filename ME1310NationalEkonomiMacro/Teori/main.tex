\subsection{Nominel vs real BNP}
Nominell BNP beräknas 
$$NGDP_t = \sum_{i=1}^{n} P_i Q_i$$

\noindent Där $P_i$ är priset på vara $i$ och $Q_i$ är kvantiteten av vara $i$ som produceras under år $t$.

\vspace{15pt}
\noindent Tillväxt i nominell BNP beräkas som: 
$$G_{t+1} = \frac{NGDP_t - NGDP_{t-1}}{NGDP_{t-1}}$$
$$G_{t+1} = \sum_{i}\omega_{i,t}(\hat{P}_{i,t+1} + \hat{Q}_{i,t+1} + \hat{P}_{i,t+1} \hat{Q}_{i,t+1})$$
$$\omega_{i,t} = \frac{P_i Q_i}{NGDP_t}$$
$\hat{}$ representerar procentuell förändring mellan år $t$ och $t+1$
\vspace{15pt}

\noindent Hur ska vi fånga upp enbart kvanitetsförändring?
$$G_{t+1} = \sum_{i}\omega_{i,t}(\hat{P}_{i,t+1} + \hat{Q}_{i,t+1} + \hat{P}_{i,t+1} \hat{Q}_{i,t+1}), \omega_{i,t} = \frac{P_i Q_i}{NGDP_t}$$

\noindent Vi håller priserna konstanta vid ett valt basår
$$g_{t+1} = \sum_{i} \hat{\omega}_{i,t} \hat{Q}_{i,t+1}, \hspace{15pt} \hat{\omega}_{i,t} = \frac{P_{i,B} Q_{i,t}}{P_{i,B} Q_{i,B}}, \hspace{15pt} RGDP_t = \sum_{i} P_{i,B} Q_{i,t}$$

\noindent $g$ är den reala tillväxten i ekonomin mellan år $t$ och $t+1$. $P_{i,B}$ är priset på vara $i$ i basåret. $RGDP$ är den reala BNPn.

\vspace{15pt}
\noindent $G$ inkluderar båda pris- och kvanitetsänderingar, medan $g$ inkluderar endast kvantitetsförändringar.

\subsection{Inflation, nominell och real ränta}
Antar att jag idag har $x$ kronor då kan jag vid tidspunkt $t$ köpa:
$$\frac{x}{P_t} \text{enheter BNP}$$
\noindent Om jag istället investerar mina pengar kan jag vid tidpunkt $t+1$ köpa: 
$$\frac{x(1+R)}{P_{t+1}} \text{enheter BNP}$$ 
\noindent Den reala räntan fås genom:
$$r_{t+1} = \frac{x(1+R_t)/P_{t+1} - x/P_t}{x/P_t} = \frac{1+R_t}{1 + \pi_{t+1}}$$
$$\implies R_t = r_{t+1} + \pi_{t+1} + r_{t+1} \pi_{t+1}$$

\subsection{Tillväxtdekomponering}
Produktionsfunktion: $Y = A \cdot F(K,N)$
\begin{itemize}
    \item $Y$: BNP
    \item $A$: Teknologisk nivå
    \item $K$: Kapitalstock
    \item $N$: Antal sysselsatta
\end{itemize}

\noindent I perfekt konkurrens gäller:
$$w = MPL = A \cdot F_N(K,N), \hspace{15pt} r = MPK = A \cdot F_K(K,N)$$

\noindent Sammanfattningvis:
$$\frac{\Delta Y}{Y} = \frac{\Delta A}{A} + \underbrace{(\frac{w \cdot N}{Y})}_{1-\theta}\frac{\Delta N}{N} + \underbrace{(\frac{r \cdot K}{Y})}_{\theta}\frac{\Delta K}{K}$$

\begin{itemize}
    \item Arbetskraftens bidgrag till tillväxt: {\Large $(\frac{w \cdot N}{Y})\frac{\Delta N}{N}$}
    \item Kapitalets bidrag till tillväxt: {\Large $(\frac{r \cdot K}{Y})\frac{\Delta K}{K}$}
\end{itemize}

\subsection{Neoklassisk produktionsfunktion, villkor}
Alla produktionsfunktioner måste uppfylla följande villkor för att användas i solowmodellen:
\begin{enumerate}
    \item Konstant skalavkastning: $F(A,\lambda K, \lambda N) = \lambda Y, \text{\hspace{15pt}} \lambda > 0$
    \item Positiva och avtagande marginalprodukter: 
    \begin{itemize}
        \item {\large $\frac{\partial F(A,K,N)}{\partial K} > 0$}
        \item {\large $\frac{\partial F(A,K,N)}{\partial N} > 0$}
        \item {\large $\frac{\partial^2 F(A,K,N)}{\partial K^2} < 0$}
        \item {\large $\frac{\partial^2 F(A,K,N)}{\partial N^2} < 0$}
    \end{itemize}
    \item Inadavillkor
    \begin{itemize}
        \item {\large $\lim\limits_{K \to 0}\frac{\partial F(A,K,N)}{\partial K} = \lim\limits_{N \to 0}\frac{\partial F(A,K,N)}{\partial N} = \infty$}
        \item {\large $\lim\limits_{K \to \infty}\frac{\partial F(A,K,N)}{\partial K} = \lim\limits_{N \to \infty}\frac{\partial F(A,K,N)}{\partial N} = 0$}
    \end{itemize} 

\end{enumerate}

\subsection{Solowmodellen}
Produktionsfunktion: $Y = A \cdot F(K,N)$
\vspace{5pt}

\noindent På grund av konstant skalavkastning kan vi skriva om produktionsfunktionen i per capita termer:
\begin{itemize}
    \item $y = f(k)$
    \item $y = \frac{Y}{N}$
    \item $k = \frac{K}{N}$
    \item $f(k) = A \cdot F(\frac{K}{N},1)$
\end{itemize}

\noindent Aggregerad kapitalachumuleringsfunktion: 
\begin{itemize}
    \item $\Delta K = sY - dK$
    \item Vi antar en sluten ekonomi dvs. en ekonomi utan utrikeshandels 
    \item Vi har inte heller in offentlig sektor, därför går alla besparingar till investeringar
\end{itemize}

\noindent Förändering i kapitalstock är alltså:
$$sparkvot * produktion - deprecieringstakt * kapitalstock$$

\noindent i per capita termer:
$$\Delta k = s y - (d + n)k, \hspace{7pt}\text{n är populations tillväxten}$$

\noindent Sammanfattningsvis två funktioner:
\begin{itemize}
    \item Produktionsfunktion: $y = f(k)$
    \item Kapitalackumuleringsfunktion: $\Delta k = s y - (d + n)k$
    \item Systems jämnvikt: $\Delta k = 0 \implies s f(k^*) = (d + n)k^*$
\end{itemize}

\end{itemize}

\subsection{Solowmodellen med teknologisk utveckling}
Produktionsfunktion: $Y = F(K, A \cdot N)$

\noindent Omskrivning med konstant skalavkastning:
\begin{itemize}
    \item $\tilde{y} = f(\tilde{k})$
    \item $\tilde{y} = \frac{Y}{A \cdot N}$
    \item $\tilde{k} = \frac{K}{A \cdot N}$
    \item $f(\tilde{k}) = F(\frac{K}{A \cdot N},1)$
\end{itemize}

\noindent Aggregerad kapitalackumuleringsfunktion: 
$$\Delta K = sY - dK$$

\noindent Uttryckt i per capitaform:
$$\Delta \tilde{k} = s \tilde{y} - (d + n + g)\tilde{k}, \hspace{7pt}\text{g är teknologisk utveckling}$$

\subsection*{Tillväxtdekomponering}
\begin{itemize}
    \item Y = BNP
    \item A = Teknologisk nivå
    \item K = Kapitalstock
    \item N = Antal sysselsatta
    \item w = Lön
    \item r = Ränta
    \item $MPL = \text{Arbetskraftens marginalproduktivitet} = \frac{\partial Y}{\partial N}$
    \item $MPK = \text{Kapitalets marginalproduktivitet} = \frac{\partial Y}{\partial K}$
    \item Produktionsfunktionen: $Y = A \cdot F(K, N)$
    \item Cobb Douglas produktionsfunktion: $Y = A \cdot K^\theta \cdot N^{1-\theta}$
\end{itemize}

\noindent I perfekt konkurrens gäller:
$$w = MPL = \frac{\partial Y}{\partial N}, \hspace{10pt} r = MPK = \frac{\partial Y}{\partial K}$$
samt att:
$$\frac{\Delta Y}{Y} = \frac{\Delta A}{A} + \underbrace{(\frac{w \cdot N}{Y})}_{1-\theta}\frac{\Delta N}{N} + \underbrace{(\frac{r \cdot K}{Y})}_{\theta}\frac{\Delta K}{K}$$

\noindent Teknologin beräknas som en residual i Solow:
$$\frac{\Delta A}{A} = \frac{\Delta Y}{Y} - (1-\theta)\frac{\Delta N}{N} - \theta \frac{\Delta K}{K}$$

\subsection*{Tillväxt teori}
\begin{itemize}
    \item Exogen tillväxteori: Teknologisk utveckling sker utanför modellen
    \item Endogen tillväxtteori: Teknologisk utvecking sker inom modellen
    \item Solowmodellen baseras på exogen tillväxtteori med en neoklassik produktionsfunktion
\end{itemize}

\subsection*{Villkor för neoklassisk produktions funktion}
\begin{enumerate}
    \item Konstant skalavkastning
    \item Positiv och avtagande marginalavkastning
    \item Inada-villkor 1: K eller N går så går partial derivatan mot oändligheten
    \item Inada-villkor 2: K eller N går mot oändligheten så går partial derivatan mot noll
\end{enumerate}

\noindent Matematiskt uttryckt:
\begin{enumerate}
    \item $F(\lambda K, \lambda N) = \lambda F(K, N)$
    \item $F_K' > 0, F_N' > 0, F_{KK}'' < 0, F_{NN}'' < 0$
    \item $\lim_{K \to 0} F_K' = \infty, \lim_{N \to 0} F_N' = \infty$
    \item $\
    lim_{K \to \infty} F_K' = 0, \lim_{N \to \infty} F_N' = 0$
\end{enumerate}

\subsection*{Solowmodellen}
\begin{itemize}
    \item Y = BNP
    \item K = Kapital
    \item N = Arbetskraft
    \item d = Depricieringstakt
    \item s = Sparkvot
    \item n = Populationstillväxttakt
\end{itemize}
Givet en produktionsfunktion: $Y = A \cdot F(K, N)$
\vspace{10pt}

\noindent Konstant skalavkastning ger omskrivningen 
$$y = f(k) = F(\frac{K}{N}, 1), \hspace{10pt} y = \frac{Y}{N}, \hspace{10pt}k = \frac{K}{N}$$

\noindent Kapitalackumulationen ges av:
$$\Delta K = sY - dK$$
\vspace{10pt}
\noindent I per capita form blir det:
$$\Delta k = sy - (d + n)k$$
\vspace{10pt}
\noindent Jämnvikt sker när $\Delta k = 0$:
$$sy^* = (d + n)k^*$$

\subsection*{Solowmodellen med teknologisk utveckling}
$$Y = A \cdot F(K, A \cdot N)$$
$$y = f(k) = F(\frac{K}{A \cdot N}, 1) = \frac{Y}{A \cdot N}, \hspace{10pt}k = \frac{K}{A \cdot N}$$

\noindent Kapitalackumulationen ges av:
$$\Delta K = sY - dK$$
\vspace{10pt}
\noindent I per capita form blir det:
$$\Delta k = sy - (d + n + g)k$$
\vspace{10pt}
$$g = \frac{\Delta y}{y} = \frac{\Delta A}{A}$$

\subsection*{Philipskurvan}
Visar ett förhållande mellan (löne-)inflation och arbetslöshet
\begin{itemize}
    \item $u =$ Arbetslöshet
    \item $u^* =$ Naturlig arbetslöshet
    \item $\epsilon =$ Känslighet
    \item W = lönenivån
\end{itemize}
$$g_w = \frac{W_{t+1} - W_t}{W_t} = \text{Förändering i lönenivå}$$
$$g_w = -\epsilon (u - u^*) = \text{Philipskurvans ekvation}$$
$$\text{Hög förändering i lönenivå} \implies \text{låg arbetslöshet}$$
$$\text{Låg förändering i lönenivå} \implies \text{hög arbetslöshet}$$

\noindent Denna samband gäller bara under kort sikt, på lång sikt är arbetslösheten naturliga arbetslösheten

\end{itemize}
Jämnvikten fås som:
$$AD_0 = Y_0 \implies \overline{A} + cY_0 = Y_0 \implies Y_0 = \frac{\overline{A}}{1 - c}$$

\subsection*{Multiplikator}
\begin{itemize}
    \item $AD$ = Aggregerad efterfrågan
    \item $c$ = Marginal konsumtionsbenägenhet (MPC)
\end{itemize}

$$\alpha = \Delta AD = \frac{1}{1+c}$$s

\subsection*{Jämnvikt i ekonomin}

\subsection*{IS-kurvan}
Visar kombinationer av räntenivå (i) och produktion/inkomst(Y) där det råder jämnvikt
\begin{itemize}
    \item Koppling mellan ränta och investeringar
    \item Koppling mellan investeringar och AD
\end{itemize}

\subsection*{IS-kurvans funktion}
Investeringsfunktion:
\begin{itemize}
    \item I = Investeringar
    \item $\overline{I}$ = Autonom investering
    \item b = Känslighet av investeringar för ränteförändringar
    \item i = Ränta
    \item $\alpha_G = \frac{1}{1-c(1-t)}$
    \item $Y = \alpha_G (\overline{A} - bi)$
\end{itemize}
$$I = \overline{I} - bi, \hspace{10pt} b > 0$$

\noindent IS-kurvan ges av:
$$i = \frac{\overline{A}}{b} - (\frac{1}{\alpha_G b})Y$$

\subsection*{LM-kurvan}
Visar kombinationer av ränta och produktionsnivå där efterfrågan på pengar motsvarar utbudet på pengar
\begin{itemize}
    \item Förklara hur efterfrågan på pengar beror på ränta och inkomst
    \item Hitta kombinationer av inkomst och ränta som ger jämvikt på penningsmarkanden
\end{itemize}

\subsection*{LM-kurvan: Efterfrågan på pengar}
Efterfrågan på pengar beror på:
\begin{itemize}
    \item Inkomst
    \item Ränta
\end{itemize}

\noindent Definieras som:
\begin{itemize}
    \item L = Efterfrågan på pengar
    \item k = Känslighet av efterfrågan på pengar för förändering i inkomst
    \item h = Känslighet av efterfrågan på pengar för förändering i ränta
\end{itemize}
$$L = kY - hi, \hspace{10pt} k > 0, h > 0$$

\subsection*{LM-kurvan: Utbudet av pengar}
Utbudet av pengar är den reala penningsmängden och beror enbart på exogen faktorer:
\begin{itemize}
    \item $\overline{M} = \text{Reala penningmängden}$
    \item $\overline{P} = \text{Prisnivån}$
\end{itemize}

$$\text{Reala penningsmängden} = \frac{\overline{M}}{\overline{P}}$$

\subsection*{LM-kurvan: Ekvation}
\begin{itemize}
    \item L = Efterfrågan på pengar
    \item M = Utbudet av pengar
    \item k = Känslighet av efterfrågan på pengar för förändering i inkomst
    \item h = Känslighet av efterfrågan på pengar för förändering i ränta
    \item $\overline{M}$ = Penningmängden
    \item $\overline{P}$ = Prisnivån
    \item $\frac{\overline{M}}{\overline{P}}$ = Reala penningmängden
    \item i = Ränta
\end{itemize}
Sätt efterfrågan på pengar lika med utbudet av pengar:
$$kY - hi = \frac{\overline{M}}{\overline{P}} \implies i = \frac{1}{h}(kY - \frac{\overline{M}}{\overline{P}})$$

\subsection*{Kopplingar till AD}
Lös ut $Y$ genom att sätta $IS = LM$:
$$\frac{\overline{A}}{b} - (\frac{1}{\alpha_G b})Y = \frac{1}{h}(kY - \frac{\overline{M}}{\overline{P}})$$
$$\implies Y = (\frac{h \alpha_G}{h + bk\alpha_G})\overline{A} + (\frac{b \alpha_G}{h + bk\alpha_G})\frac{\overline{M}}{\overline{P}}$$e
